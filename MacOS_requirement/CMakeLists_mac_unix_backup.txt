cmake_minimum_required(VERSION 3.10)
project(location_tracker)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(location_tracker
    main.cpp
    locationTracker.cpp
)

target_include_directories(location_tracker PRIVATE .)

find_package(Boost REQUIRED)

if(Boost_FOUND)
    message(STATUS "Boost found at: ${Boost_INCLUDE_DIRS}")
    target_include_directories(location_tracker PRIVATE ${Boost_INCLUDE_DIRS})
else()
    message(WARNING "Boost not found system-wide. Make sure Boost headers are installed.")
endif()

target_link_libraries(location_tracker PRIVATE pthread)
